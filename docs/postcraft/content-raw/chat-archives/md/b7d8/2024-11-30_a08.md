 1. Hot reloading with Docker volumes:
```docker-compose.yml
services:
  app:
    build: .
    volumes:
      - .:/app
      - /app/node_modules
      - /app/packages/local-pkg/node_modules
    command: npm run dev
```
Use nodemon or similar for auto-restart:
```json
{
  "scripts": {
    "dev": "nodemon --legacy-watch src/index.js"
  }
}
```

4. Development docker-compose setup:
```yaml
services:
  app:
    build: 
      context: .
      target: development
    volumes:
      - .:/app
    environment:
      - NODE_ENV=development
    ports:
      - "3000:3000"
      - "9229:9229"  # Debug port
    depends_on:
      - db
  db:  # Example dependency
    image: postgres:13
    environment:
      - POSTGRES_DB=devdb
```
