services:
  app:
    build: .
    ports:
      - "3000:3000"
    depends_on:
      - fuseki
    environment:
      - FUSEKI_URL=http://fuseki:3030
      - FUSEKI_DATASET=ds

  fuseki:
    image: stain/jena-fuseki
    ports:
      - "3030:3030"
    environment:
      - ADMIN_PASSWORD=your_admin_password
      - JVM_ARGS=-Xmx2g
    volumes:
      - ./fuseki-data:/fuseki
      - ./config.ttl:/fuseki/config.ttl
    command: "--config=/fuseki/config.ttl"

  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf:ro
    depends_on:
      - app
      - fuseki
</antArtifact>

<antArtifact identifier="fuseki-nginx-config" type="application/vnd.ant.code" language="nginx" title="Nginx Configuration for Fuseki">
events {
    worker_connections 1024;
}

http {
    upstream app_server {
        server app:3000;
    }

    upstream fuseki_server {
        server fuseki:3030;
    }

    server {
        listen 80;
        
        location / {
            proxy_pass http://app_server;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
        }

        location /fuseki/ {
            proxy_pass http://fuseki_server/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
        }
    }
}
</antArtifact>
