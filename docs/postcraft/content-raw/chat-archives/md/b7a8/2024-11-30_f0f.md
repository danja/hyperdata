 No - we need to align the ports. Currently:

Docker compose has:
- External 4030 -> Internal 3030 (Fuseki)
- External 4080 -> Internal 80 (Nginx)

But nginx.conf uses ports 3770 and 3773.

Let's modify nginx.conf to listen on port 80:

<antArtifact identifier="aligned-nginx" type="application/vnd.ant.code" language="nginx" title="Aligned Nginx Configuration">
events {
    worker_connections 1024;
}

http {
    upstream app_server {
        server app:3000;
    }

    upstream fuseki_server {
        server fuseki:3030;
    }

    server {
        listen 80;
        
        location / {
            proxy_pass http://app_server;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
        }

        location /fuseki/ {
            proxy_pass http://fuseki_server/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
        }
    }
}
</antArtifact>
