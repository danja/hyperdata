        class Organism {
            constructor(x, y) {
                this.x = x;
                this.y = y;
                this.width = Math.random() * 20 + 30;
                this.height = this.width * 0.6;
                this.dx = (Math.random() - 0.5) * 2;

                this.color = this.generateColor();

            update() {

                    if (newX > rect.left - this.width/2 && newX < rect.right + this.width/2 &&
